	d) Mostrar el nombre de los departamentos de una sucursal ... Coacalco

		// Consulta de más de dos relaciones, relaciones-1 = numero de selecciones
		Select h.nombre, d.nombre
		from homedepot h, depto d, hddepto x
		where h.idhd=x.homedepot_idhd
		and x.depto_iddepto=d.iddepto
		and h.nombre like “Coacalco%"
		order by 1;

3.- Mostrar el nombre de las sucursales y sus asociados, de aquellas ubicadas en las siguientes cp 78045, 89360, 25279

	Select h.nombre, a.nombre
	from homedepot h, asociado a

5.- Qué teléfono tienen los socios que tienen la tarjeta Pagos fijos y mostrar la sucursal donde se encuentran dados de alta

	// 4 relaciones, 3 atributos de reunión - a pesar de ser sólo dos campos
	Select s.nombre, s.tel
	from socio s, tarjeta t, hdsocio x, homedepot h // importa el orden, revisar
	where s.idsocio=t.socio_idsocio // seguir camino en las relaciones, si no lo seguimos, se produce un producto cartesiano... no acaba
	and s.idsocio=x.socio_idsocio
	and x.homedepot_idhd=h.idhd
	and t.nombre like "Pag% fij%"
	order by 1;

6.- Mostrar los departamentos que tiene la sucursal tijuana

	Select h.nombre as "Tienda", d.nombre as "Departamento"
	from homedepot h, depto d, hddepto x
	where h.idhd=x.homedepot_idhd
	and x.depto_iddepto=d.iddepto
	and h.nombre like "Tiju%"
	order by 2 desc;
