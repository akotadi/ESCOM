create view v1 as
Select m.idmixup as IDENTIFICADOR, s.nombre as NOMBRE
from mixup m, sociomix sm, socio s
where m.idmixup=sm.idmixup
and sm.idsocio=s.idsocio
order by 1,2;

create view v2 as
Select m.idmixup as IDENTIFICADOR, sm.idsocio as IDSOCIO
from mixup m, sociomix sm
where m.idmixup=sm.idmixup
order by 1,2;

create view v3 as
select idarticulo, nombre
from articulo
order by 1; 

create view v4 as

create view v5 as

create view v6 as
Select m.nombre as MIXUP, s.idsocio as IDSOCIO
from mixup m, sociomix sm, socio s
where m.idmixup=sm.idmixup
and sm.idsocio=s.idsocio
order by 1,2;

create view v7 as

create view v8 as

create view v9 as

create view v10 as
select a.idarticulo, d.nombre as DEPARTAMENTO
from articulo a, departamento d
where a.iddepto=d.iddepto
order by 2,1;

create view v11 as
select nombre, ubicacion, dir
from mixup
order by 1;
